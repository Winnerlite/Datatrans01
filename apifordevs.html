Sample Integration

const vyra = require('vyra-sdk');

// 1. Find cheapest bus to Enugu
async function getCheapestRide() {
const options = await vyra.search({
from: 'Lagos',
to: 'Enugu',
sort: 'price_asc'
});

return options[0];
}
